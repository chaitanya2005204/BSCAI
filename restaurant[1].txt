  <script>
    // Sample data - update or replace with dynamic data as needed
    const restaurants = [
      { name: "Tandoori Flames", cuisine: "Indian", location: "Visakhapatnam", rating: 4.5, contact: "9876543210", image: "1.jpg" },
      { name: "Sushi Express", cuisine: "Japanese", location: "Mumbai", rating: 4.2, contact: "9876501234", image: "2.jpg" },
      { name: "Burger Bites", cuisine: "American", location: "Delhi", rating: 4.1, contact: "9867543210", image: "3.jpeg" },
      { name: "Noodle House", cuisine: "Chinese", location: "Hyderabad", rating: 4.3, contact: "9898989898", image: "4.webp" },
      { name: "Pizza Mania", cuisine: "Italian", location: "Bangalore", rating: 4.0, contact: "9123456789", image: "5.jpeg" }
    ];

    const locationInput = document.getElementById('locationInput');
    const cuisineInput = document.getElementById('cuisineInput');
    const restaurantList = document.getElementById('restaurantList');

    function createCard(restaurant) {
      const card = document.createElement('div');
      card.className = 'card';
      // If image file missing, browser will show broken image - you can replace with placeholder URL
      card.innerHTML = `
        <img src="${restaurant.image}" alt="${restaurant.name} photo" onerror="this.onerror=null;this.src='https://via.placeholder.com/600x400?text=No+Image'">
        <div class="card-content">
          <h3>${restaurant.name}</h3>
          <div class="meta">
            <span><strong>Cuisine:</strong> ${restaurant.cuisine}</span>
            <span><strong>Location:</strong> ${restaurant.location}</span>
            <span class="rating">‚≠ê ${restaurant.rating.toFixed(1)}</span>
          </div>
          <div class="contact"><strong>Contact:</strong> üìû ${restaurant.contact}</div>
        </div>
      `;
      return card;
    }

    function displayRestaurants(list) {
      restaurantList.innerHTML = '';
      if (list.length === 0) {
        restaurantList.innerHTML = '<p style="text-align:center;color:#666">No restaurants found.</p>';
        return;
      }
      list.forEach(r => restaurantList.appendChild(createCard(r)));
    }

    function filterRestaurants() {
      const loc = locationInput.value.trim().toLowerCase();
      const cuisine = cuisineInput.value.trim().toLowerCase();
      const filtered = restaurants.filter(r =>
        r.location.toLowerCase().includes(loc) &&
        r.cuisine.toLowerCase().includes(cuisine)
      );
      displayRestaurants(filtered);
    }

    locationInput.addEventListener('input', filterRestaurants);
    cuisineInput.addEventListener('input', filterRestaurants);

    // initial render
    displayRestaurants(restaurants);
  </script>
